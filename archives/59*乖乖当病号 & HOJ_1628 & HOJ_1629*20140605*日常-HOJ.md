    #include <iostream>
    using namespace std;
    
    int main(){
        int n, nums[100], half, sum, pos;
        bool eql;
        while(cin >> n && n){
            sum = half = 0;
            for(int i = 1; i <= n; i++){
                cin >> nums[i];
                sum += nums[i];
            }
            if(sum % 2 == 1)
                cout << "No equal partitioning." << endl;
            else{
                eql = false;
                for(int i = 1; i <= n; i++){
                    half += nums[i];
                    if(half == sum / 2){
                        pos = i;
                        eql = true;
                        break;
                    }
                }
                if(eql)
                    cout << "Sam stops at position " << pos << " and Ella stops at position " << pos + 1 << '.' << endl;
                else
                    cout << "No equal partitioning." << endl;
            }
        }
    }
