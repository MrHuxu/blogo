1526：非常水的一道题，就是因数分解，因为今天晚上和爸妈Facetime了，就做了这么一道简单的。  

代码如下：  

    #include <iostream>
    #include <math.h>
    using namespace std;
    
    int main(){
        long long input;
        while(cin >> input && input > 0){
            for(int i = 2; i <= (int)sqrt((double)input); i++){
                for(; !(input % i); input /= i)
                    cout << i << endl;
            }
            if(input > 1)
                cout << input << '\n' << endl;
        }
    }
